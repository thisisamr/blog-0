import NextImage from 'next/image'
import {useState,useEffect} from 'react'
import {useTheme} from 'next-themes'
export default function Home() {
  const [mounted,setMounted]= useState(false)
  const {resolvedTheme} = useTheme()
  useEffect(()=>{
    setMounted(true)
  },[])
  if(!mounted){
    return null
  }
  return (
    <div className="flex flex-col md:flex-row  md:items-center max-w-screen-sm md:max-w-full">
    <NextImage priority src={resolvedTheme=='light'?'/hero_light.png':'/hero_dark.png'} width= {800} height={800}alt='hero' fetchPriority='auto'/>
    <div>
      <p className=" uppercase tracking tracking-widest inter text-sm w-[300px] bg-gradient-to-r dark:from-purple-600 from-black via-gray-800 to-slate-600 dark:via-cyan-600 dark:to-pink-800  max-w-xs text-transparent bg-clip-text ">
Prepare to lose yourself in the haunting depths of my mind.      </p>

    </div>
    </div>
  )
}
